### 二、文件读写（通用模块API）
#### 1. 保存文件（SaveFile函数）
```cpp
bool SaveFile(LPCTSTR szFileName) {
    // 打开文件
    HANDLE hFile = EFOpen(szFileName, GENERIC_WRITE);
    if (hFile == INVALID_HANDLE_VALUE) return false;

    // 向文件写入文本框内容（EF_LineSeed_None表示不额外添加换行符）
    EFPrint(hFile, form1.Control(ID_txtMain).Text(), EF_LineSeed_None);
    
    EFClose(hFile);  // 关闭文件
    return true;
}
```

#### 2. 按行读取文件（CBReadLinesEx类）
需先添加 `BReadLinesEx.cpp`、`BReadLinesEx.h` 到项目。
```cpp
#include "BReadLinesEx.h"

bool OpenFile(LPCTSTR szFileName) {
    tstring sContent(TEXT(""));  // 存储文件内容
    CBReadLinesEx file;  // 定义按行读取文件的对象
    
    if (!file.OpenFile(szFileName)) return false;  // 打开文件
    pApp->MousePointerGlobalSet(IDC_Wait);  // 设置鼠标为沙漏状

    while (!file.IsEndRead()) {  // 循环读取每一行
        LPTSTR szLine = NULL;
        file.GetNextLine(szLine);  // 获取下一行内容
        if (file.IsErrOccured()) {
            pApp->MousePointerGlobalSet(0);
            return false;
        }

        // 拼接内容（统一换行符为\r\n）
        if (file.iEndLineSignLast) sContent += TEXT("\r\n");
        sContent += szLine;
    }

    // 将内容显示到文本框
    form1.Control(ID_txtMain).TextSet(sContent);
    _tcscpy(m_szFileName, szFileName);  // 保存文件名到全局变量
    form1.TextSet(m_szFileName);  // 窗体标题栏显示文件名
    form1.TextAdd(TEXT("-我的记事本"));
    pApp->MousePointerGlobalSet(0);  // 恢复鼠标
    m_b1ModiSign = false;  // 设置“未修改”标志
    return true;
}
```

bool QuerySave() {
    if (m_b1ModiSign) {  // 若内容已修改
        EDlgBoxCmdID ms = MsgBox(TEXT("您已经对内容做了修改。\r\n\r\n是否保存文件?"),
                                  TEXT("我的记事本"), mb_YesNoCancel, mb_IconExclamation);
        if (ms == idYes) {
            Menu_Click(ID_mnuFileSave, false, false);  // 模拟点击“保存”
            if (m_b1ModiSign) return true;  // 若保存失败，阻止后续操作
        } else if (ms == idCancel) {
            return true;  // 用户取消，阻止后续操作
        }
    }
    return false;  // 可以继续操作
}
